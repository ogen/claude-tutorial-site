    第三十章：台股分析 Agent 完整實作¶

    這章要做什麼？¶

    前面九個模組學了 Claude API、工具呼叫、ReAct 框架、Multi-agent、排程自動化——這章把它們全部整合在一個真實的專案裡：一個能自動分析台股的 AI Agent。

    你告訴它「分析台積電」，它會自己去抓股價資料、計算技術指標、查財務數字，然後整合成一份完整的投資分析報告。

    免責聲明：本章內容純屬技術教學，不構成任何投資建議。股票投資有風險，請自行判斷。

    架構設計¶

    整個系統分三層：

    使用者問題（自然語言）
        │
        ▼
  Stock Analysis Agent（ReAct 循環）
        │
        ├── get_stock_price()        ← 取得歷史股價
        ├── get_financial_metrics()  ← 財務指標（本益比、EPS、ROE）
        └── calculate_technical()    ← 技術分析（MA、RSI、MACD）
        │
        ▼
   整合分析報告（繁體中文）

    如果要同時分析多支股票，再套一層 Multi-agent：

    Orchestrator Agent
    ├── Sub-agent A → 分析 2330（台積電）
    ├── Sub-agent B → 分析 2317（鴻海）
    └── Sub-agent C → 分析 0050（元大台灣 50）
         ↓
  三份報告整合 → 比較摘要

    環境設定¶

    這個專案需要的套件不多，先安裝起來：

    pip install anthropic yfinance pandas

    yfinance 是 Yahoo Finance 的 Python 封裝，台股代號格式是「數字代號加 .TW」，例如台積電是 2330.TW。

    建立專案目錄：

    mkdir taiwan-stock-agent
cd taiwan-stock-agent
touch agent.py

    設定 API Key（若還沒設定過）：

    export ANTHROPIC_API_KEY="sk-ant-..."

    Windows 用戶：用 set ANTHROPIC_API_KEY=sk-ant-... 或在 PowerShell 用 $env:ANTHROPIC_API_KEY="sk-ant-..."

    工具函數實作¶

    定義好工具 Schema 之後，要實作真正執行資料抓取的函數。這裡用 yfinance 連到 Yahoo Finance 取得台股資料：

    # tools_impl.py
import yfinance as yf
import pandas as pd
from datetime import datetime


def get_stock_price(ticker: str, period: str) -> dict:
    """取得台股歷史股價"""
    tw_ticker = f"{ticker}.TW"
    stock = yf.Ticker(tw_ticker)
    hist = stock.history(period=period)

    if hist.empty:
        return {"error": f"無法取得 {ticker} 的股價資料，請確認代號是否正確"}

    current_price = hist["Close"].iloc[-1]
    start_price = hist["Close"].iloc[0]
    price_change_pct = (current_price - start_price) / start_price * 100
    period_high = hist["Close"].max()
    period_low = hist["Close"].min()
    avg_volume = int(hist["Volume"].mean())

    return {
        "ticker": ticker,
        "query_date": datetime.today().strftime("%Y-%m-%d"),
        "period": period,
        "current_price": round(current_price, 2),
        "period_start_price": round(start_price, 2),
        "price_change_pct": round(price_change_pct, 2),
        "period_high": round(period_high, 2),
        "period_low": round(period_low, 2),
        "avg_daily_volume": avg_volume,
    }


def get_financial_metrics(ticker: str) -> dict:
    """取得財務基本面指標"""
    tw_ticker = f"{ticker}.TW"
    stock = yf.Ticker(tw_ticker)
    info = stock.info

    if not info or info.get("regularMarketPrice") is None:
        return {"error": f"無法取得 {ticker} 的財務資料"}

    # 股息殖利率轉為百分比
    div_yield = info.get("dividendYield")
    div_yield_pct = round(div_yield * 100, 2) if div_yield else "N/A"

    # ROE 轉為百分比
    roe = info.get("returnOnEquity")
    roe_pct = round(roe * 100, 2) if roe else "N/A"

    return {
        "ticker": ticker,
        "company_name": info.get("longName", "未知"),
        "sector": info.get("sector", "未知"),
        "pe_ratio": round(info.get("trailingPE", 0), 2) or "N/A",
        "forward_pe": round(info.get("forwardPE", 0), 2) or "N/A",
        "eps_ttm": info.get("trailingEps", "N/A"),
        "roe_pct": roe_pct,
        "revenue_growth_yoy": info.get("revenueGrowth", "N/A"),
        "dividend_yield_pct": div_yield_pct,
        "market_cap_twd": info.get("marketCap", "N/A"),
        "52_week_high": info.get("fiftyTwoWeekHigh", "N/A"),
        "52_week_low": info.get("fiftyTwoWeekLow", "N/A"),
    }


def calculate_technical_indicators(ticker: str) -> dict:
    """計算技術分析指標"""
    tw_ticker = f"{ticker}.TW"
    stock = yf.Ticker(tw_ticker)
    hist = stock.history(period="1y")

    if hist.empty or len(hist) < 20:
        return {"error": f"資料不足，無法計算 {ticker} 的技術指標"}

    close = hist["Close"]
    current = round(close.iloc[-1], 2)

    # 移動平均線
    ma20 = round(close.rolling(20).mean().iloc[-1], 2)
    ma50 = round(close.rolling(50).mean().iloc[-1], 2) if len(hist) >= 50 else None
    ma200_val = close.rolling(200).mean().iloc[-1]
    ma200 = round(ma200_val, 2) if not pd.isna(ma200_val) else None

    # RSI（14 日）
    delta = close.diff()
    gain = delta.where(delta > 0, 0).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    rs = gain / loss
    rsi = round((100 - 100 / (1 + rs)).iloc[-1], 2)

    # MACD（12/26/9）
    ema12 = close.ewm(span=12).mean()
    ema26 = close.ewm(span=26).mean()
    macd_line = ema12 - ema26
    signal_line = macd_line.ewm(span=9).mean()
    macd_val = round(macd_line.iloc[-1], 2)
    signal_val = round(signal_line.iloc[-1], 2)

    return {
        "ticker": ticker,
        "current_price": current,
        "ma20": ma20,
        "ma50": ma50,
        "ma200": ma200,
        "rsi_14": rsi,
        "macd": macd_val,
        "macd_signal": signal_val,
        "ma20_trend": "價格在 MA20 之上（短期偏多）" if current > ma20 else "價格在 MA20 之下（短期偏空）",
        "ma50_trend": ("價格在 MA50 之上（中期偏多）" if ma50 and current > ma50 else "價格在 MA50 之下（中期偏空）"),
        "rsi_signal": (
            "超買區間（>70），注意短線回調風險" if rsi > 70
            else "超賣區間（<30），留意反彈機會" if rsi < 30
            else f"中性區間（{rsi}）"
        ),
        "macd_signal_trend": "MACD 在訊號線之上（動能偏多）" if macd_val > signal_val else "MACD 在訊號線之下（動能偏空）",
    }

    Agent 主程式¶

    有了工具之後，把它們接進 ReAct Agent 循環。Claude 會自己決定要呼叫哪些工具、呼叫幾次，直到它認為資訊足夠再給出最終分析：

    # agent.py
import anthropic
import json
from tools import STOCK_TOOLS
from tools_impl import get_stock_price, get_financial_metrics, calculate_technical_indicators


SYSTEM_PROMPT = """你是一位專業的台股分析師，擅長結合基本面、技術面進行個股分析。

分析流程：
1. 取得近 3-6 個月股價走勢，了解近期趨勢
2. 查看財務指標，評估基本面健康度
3. 計算技術指標，判斷進出場時機
4. 整合三方面資料，撰寫客觀分析報告

報告格式：
## [公司名稱]（[代號]）分析報告

### 近期走勢
[股價變化摘要]

### 基本面
[財務指標解讀，重點說明 PE、EPS、ROE]

### 技術面
[MA 趨勢、RSI、MACD 解讀]

### 綜合評估
[整合三方面的客觀評估，條列潛在機會與風險]

---
⚠️ 本報告僅供參考，不構成投資建議。"""


def execute_tool(tool_name: str, tool_input: dict) -> str:
    """執行工具並回傳 JSON 字串"""
    dispatch = {
        "get_stock_price": get_stock_price,
        "get_financial_metrics": get_financial_metrics,
        "calculate_technical_indicators": calculate_technical_indicators,
    }
    fn = dispatch.get(tool_name)
    if not fn:
        return json.dumps({"error": f"未知工具：{tool_name}"}, ensure_ascii=False)

    result = fn(**tool_input)
    return json.dumps(result, ensure_ascii=False, indent=2)


def analyze_stock(query: str, verbose: bool = True) -> str:
    """執行台股分析 Agent，回傳最終分析報告"""
    client = anthropic.Anthropic()
    messages = [{"role": "user", "content": query}]

    while True:
        response = client.messages.create(
            model="claude-sonnet-4-6",
            max_tokens=4096,
            system=SYSTEM_PROMPT,
            tools=STOCK_TOOLS,
            messages=messages,
        )

        # 沒有工具呼叫 → 分析完成，回傳最終文字
        if response.stop_reason == "end_turn":
            final_text = ""
            for block in response.content:
                if hasattr(block, "text"):
                    final_text += block.text
            return final_text

        # 把 AI 的回應加入對話歷史
        messages.append({"role": "assistant", "content": response.content})

        # 執行所有工具呼叫
        tool_results = []
        for block in response.content:
            if block.type != "tool_use":
                continue

            if verbose:
                print(f"  → 呼叫工具：{block.name}({block.input})")

            result_str = execute_tool(block.name, block.input)
            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": result_str,
            })

        # 把工具結果加回對話，讓 Claude 繼續
        messages.append({"role": "user", "content": tool_results})


if __name__ == "__main__":
    print("=== 台股分析 Agent ===\n")
    query = "請分析台積電（2330）的近期表現，從基本面和技術面給我一份完整評估。"
    print(f"問題：{query}\n")
    print("正在分析中...\n")
    report = analyze_stock(query)
    print(report)

    Multi-agent：同時分析多支股票¶

    單支股票分析夠了，如果要比較多支股票，用 asyncio 讓多個 Agent 平行執行：

    # multi_agent.py
import asyncio
import anthropic
import json
from tools import STOCK_TOOLS
from tools_impl import get_stock_price, get_financial_metrics, calculate_technical_indicators


def execute_tool(tool_name: str, tool_input: dict) -> str:
    dispatch = {
        "get_stock_price": get_stock_price,
        "get_financial_metrics": get_financial_metrics,
        "calculate_technical_indicators": calculate_technical_indicators,
    }
    fn = dispatch.get(tool_name)
    result = fn(**tool_input) if fn else {"error": f"未知工具：{tool_name}"}
    return json.dumps(result, ensure_ascii=False)


async def analyze_single_stock(ticker: str, client: anthropic.Anthropic) -> dict:
    """非同步分析單支股票（在執行緒池中執行同步 API 呼叫）"""
    import concurrent.futures

    def _run_sync():
        messages = [{"role": "user", "content": f"分析台股 {ticker}，給我近 3 個月走勢、基本面和技術面摘要。"}]
        system = "你是台股分析師，簡潔地分析給定股票，重點列出 3 個投資亮點和 2 個風險。用繁體中文回答。"

        while True:
            response = client.messages.create(
                model="claude-haiku-4-5-20251001",  # 子任務用 Haiku 節省成本
                max_tokens=1024,
                system=system,
                tools=STOCK_TOOLS,
                messages=messages,
            )

            if response.stop_reason == "end_turn":
                text = "".join(b.text for b in response.content if hasattr(b, "text"))
                return {"ticker": ticker, "analysis": text}

            messages.append({"role": "assistant", "content": response.content})
            tool_results = [
                {
                    "type": "tool_result",
                    "tool_use_id": b.id,
                    "content": execute_tool(b.name, b.input),
                }
                for b in response.content if b.type == "tool_use"
            ]
            messages.append({"role": "user", "content": tool_results})

    loop = asyncio.get_event_loop()
    with concurrent.futures.ThreadPoolExecutor() as pool:
        return await loop.run_in_executor(pool, _run_sync)


async def compare_stocks(tickers: list[str]) -> str:
    """平行分析多支股票，再由 Orchestrator 整合比較"""
    client = anthropic.Anthropic()
    print(f"同時分析 {len(tickers)} 支股票：{', '.join(tickers)}\n")

    # 平行執行所有子任務
    tasks = [analyze_single_stock(t, client) for t in tickers]
    results = await asyncio.gather(*tasks)

    # Orchestrator：整合所有子報告
    combined = "\n\n".join(
        f"### {r['ticker']} 分析結果\n{r['analysis']}" for r in results
    )
    orchestrator_prompt = f"""以下是 {len(tickers)} 支台股的個別分析結果：

{combined}

請整合以上資料，撰寫一份「比較摘要」，說明：
1. 哪支股票目前基本面最穩健？
2. 哪支股票技術面訊號最強？
3. 整體市場訊號（這幾支股票共同呈現什麼趨勢）？"""

    summary_response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=1024,
        messages=[{"role": "user", "content": orchestrator_prompt}],
    )
    return summary_response.content[0].text


if __name__ == "__main__":
    tickers = ["2330", "2317", "0050"]  # 台積電、鴻海、元大台灣 50
    summary = asyncio.run(compare_stocks(tickers))
    print("\n=== 比較摘要 ===\n")
    print(summary)

    自動化每日報告¶

    把 Agent 包成排程，每天早上自動執行並寄出報告：

    # scheduler.py
import schedule
import time
from agent import analyze_stock
from datetime import datetime


WATCHLIST = ["2330", "2317", "2454", "0050"]  # 自選清單


def generate_daily_report():
    today = datetime.today().strftime("%Y-%m-%d")
    print(f"\n[{today}] 開始產生每日台股報告...\n")

    reports = []
    for ticker in WATCHLIST:
        print(f"  分析 {ticker}...")
        analysis = analyze_stock(
            f"分析台股 {ticker}，重點說明今日值得關注的訊號。",
            verbose=False
        )
        reports.append(f"## {ticker}\n{analysis}")

    full_report = f"# 台股每日報告 — {today}\n\n" + "\n\n---\n\n".join(reports)

    # 儲存報告
    filename = f"report_{today}.md"
    with open(filename, "w", encoding="utf-8") as f:
        f.write(full_report)
    print(f"\n報告已儲存：{filename}")


# 每個交易日早上 8:30 執行
schedule.every().monday.at("08:30").do(generate_daily_report)
schedule.every().tuesday.at("08:30").do(generate_daily_report)
schedule.every().wednesday.at("08:30").do(generate_daily_report)
schedule.every().thursday.at("08:30").do(generate_daily_report)
schedule.every().friday.at("08:30").do(generate_daily_report)

if __name__ == "__main__":
    print("台股報告排程已啟動，等待下次執行時間...")
    generate_daily_report()  # 先執行一次測試
    while True:
        schedule.run_pending()
        time.sleep(60)

    Windows 用戶：schedule 套件在 Windows 上運作正常。若要改用 Windows Task Scheduler，可以把 generate_daily_report() 包成獨立執行的 Python 腳本，再用 Task Scheduler 設定每天定時執行。

    常見問題 Q&A¶

    Q：yfinance 取得的台股資料準確嗎？ A：yfinance 資料來源是 Yahoo Finance，台股基本股價資料（收盤價、成交量）通常準確，但部分財務指標（尤其是最新一季的 EPS）可能有延遲，或因台灣財報揭露時間而出現短暫的 N/A 值。若需要高精準度財務資料，可以考慮接 TEJ（台灣經濟新報）或財政部的 API，再實作對應的工具函數替換掉 yfinance 版本即可，Agent 的其餘邏輯不需更動。

    Q：Token 費用大概多少？ A：分析一支股票大約需要 3-4 次工具呼叫，整體對話約 3,000-5,000 tokens。用 claude-sonnet-4-6 計算，一次分析約 $0.02 USD；如果子任務改用 Haiku，可以降到 $0.005 以下。一份包含 5 支股票的每日報告，成本大約在 $0.05-0.10 USD 之間。

    Q：Agent 回答有時品質不穩定，怎麼辦？ A：幾個改善方向：一是在 System Prompt 加入更具體的格式要求（如「每個段落至少說明數值來源」）；二是在工具回傳結果中加入更多脈絡（如同時回傳「本益比 20 倍，高於同業平均 15 倍」而非只回傳數字）；三是對最終輸出加上後處理驗證，若報告長度異常短，自動重新呼叫。
